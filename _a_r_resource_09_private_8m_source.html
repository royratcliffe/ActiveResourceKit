00030 NSNumber *ARIDFromResponse(ARHTTPResponse *response)
00031 {
00032     NSString *location = [[response headerFields] objectForKey:@"Location"];
00033     if (location == nil)
00034     {
00035         return nil;
00036     }
00037     NSRegularExpression *re = [NSRegularExpression regularExpressionWithPattern:@"\\/([^\\/]*?)(\\.\\w+)?$" options:0 error:NULL];
00038     NSTextCheckingResult *match = [re firstMatchInString:location options:0 range:NSMakeRange(0, [location length])];
00039     if (match == nil || [match numberOfRanges] < 2)
00040     {
00041         return nil;
00042     }
00043     NSString *string = [location substringWithRange:[match rangeAtIndex:1]];
00044     NSNumberFormatter *numberFormatter = [[NSNumberFormatter alloc] init];
00045     return [numberFormatter numberFromString:string];
00046 }
